#!/bin/bash

ver=$(echo "$1" \
      |sed 's/i[3-6]86\|x86[_-]64\|x64\|[Xx][Ff][_-]\?86\|[Qq][Tt]4/masq/' \
      |grep -oE '[_-]([A-Za-z]{0,2}[0-9][a-z0-9]*[_.+-])+')

re1='\(.*\)'; re2='\1'; n=1
for partver in $ver ; do
    n=$(($n+1))
    partver=${partver%?}
    partver=${partver#?}
    re1="${re1}$partver\\(.*\\)"
    re2="${re2}.*\\$n"
    sver="${sver}${sver:+-}$partver"
done

echo "$1"
echo "$1" |sed 's/'$re1'\.pfs/'$re2'\\.pfs/'
echo $sver
echo ---------------------------------
